const express = require('express');
const { createServer } = require('http');
const { Server } = require('socket.io');
const SessionManager = require('./sessionManager');
const { 
  handleCreateSession, 
  handleJoinSession, 
  handleLeaveSession,
  handleSyncRole,
  handleRequestFullSync,
  handleKickUser,
  handleSyncPatch,
  handleTextureUpload,
  handleTextureRequest,
  handleBroadcastFullState,
  handleEphemeral
} = require('./eventHandlers');

const app = express();
const httpServer = createServer(app);
const io = new Server(httpServer, {
  cors: {
    origin: process.env.CLIENT_ORIGIN || "http://localhost:8080",
    methods: ["GET", "POST"],
    credentials: true
  },
  maxHttpBufferSize: 10 * 1024 * 1024 // 10MB for texture uploads
});

const sessionManager = new SessionManager();

// Basic health check endpoint
app.get('/health', (req, res) => {
  const textureStats = sessionManager.getTextureStats();
  res.json({ 
    status: 'ok', 
    sessions: sessionManager.getActiveSessions(),
    connections: io.engine.clientsCount,
    textures: textureStats
  });
});

io.on('connection', (socket) => {
  console.log(`[Connection] New connection: ${socket.id}`);

  // Session Management
  socket.on('create_session', (payload) => 
    handleCreateSession(socket, io, sessionManager, payload)
  );

  socket.on('join_session', (payload) => 
    handleJoinSession(socket, io, sessionManager, payload)
  );

  socket.on('leave_session', () => 
    handleLeaveSession(socket, io, sessionManager)
  );

  // Role Assignment (still needed for DM permissions)
  socket.on('sync_role', (payload) => 
    handleSyncRole(socket, io, sessionManager, payload)
  );

  // JSON Patch Sync (unified sync system for all stores)
  socket.on('sync:patch', (payload) => 
    handleSyncPatch(socket, io, sessionManager, payload)
  );

  // Texture Sync
  socket.on('texture:upload', (payload) =>
    handleTextureUpload(socket, io, sessionManager, payload)
  );

  socket.on('texture:request', (payload) =>
    handleTextureRequest(socket, io, sessionManager, payload)
  );

  // Additional Events
  socket.on('request_full_sync', () => 
    handleRequestFullSync(socket, sessionManager)
  );

  socket.on('kick_user', (payload) => 
    handleKickUser(socket, io, sessionManager, payload)
  );

  // Full state broadcast from DM
  socket.on('broadcast_full_state', (payload) =>
    handleBroadcastFullState(socket, io, sessionManager, payload)
  );

  // Ephemeral messages (cursors, drag previews, typing, etc.)
  // Broadcast only — no logging, no persistence, no sequence tracking
  socket.on('ephemeral', (payload) =>
    handleEphemeral(socket, io, sessionManager, payload)
  );

  // Disconnection handling
  socket.on('disconnect', () => {
    console.log(`[Disconnect] Client disconnected: ${socket.id}`);
    handleLeaveSession(socket, io, sessionManager);
  });
});

const PORT = process.env.PORT || 3001;
httpServer.listen(PORT, () => {
  console.log(`[Server] Socket.io server running on port ${PORT}`);
  console.log(`[Server] Accepting connections from: ${process.env.CLIENT_ORIGIN || "http://localhost:8080"}`);
});
